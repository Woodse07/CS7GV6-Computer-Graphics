#version 330

in vec3 vertexPosition;
in vec3 vertexNormals;

out vec3 nEye;
out vec3 vsNormals;
out vec3 vsPosition;

uniform mat4 view;
uniform mat4 proj;
uniform mat4 ortho;
uniform mat4 model;

void main(){
	nEye = (view * vec4 (vertexNormals, 0.0)).xyz;
	gl_Position =  proj * view * model * ortho* vec4 (vertexPosition, 1.0);
	vsNormals = vertexNormals;
	vsPosition = vertexPosition;
}
