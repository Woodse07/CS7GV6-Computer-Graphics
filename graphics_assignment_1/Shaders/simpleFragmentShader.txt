#version 330

in vec3 nEye;
in vec3 vsNormals;
in vec3 vsPosition;

out vec4 fragColour;

uniform vec3 lightPos;

void main(){
	vec3 lightVector = normalize(lightPos - vsPosition);
	float brightness = dot(lightVector, vsNormals);
	float ambientStrength = 1.0;
	fragColour = vec4 (nEye * brightness, 1.0);
}
